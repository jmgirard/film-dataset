library(tidyverse)
library(readxl)

Prepare continuous (CARMA) ratings

# Find all rating files
carma_files <- 
  dir(
    path = "./data/Carma_Rating_Results", 
    pattern = "*.xlsx", 
    full.names = TRUE
  )
# Read in all rating files
carma_raw <- 
  map_dfr(
    .x = carma_files, 
    .f = read_xlsx, 
    skip = 7, 
    .name_repair = ~str_remove_all(., "[:space:]"), 
    .id = "Video"
  ) |> 
  select(
    Video, 
    Second, 
    starts_with("MW")
  ) |> 
  mutate(
    Video = parse_integer(Video),
    Second = as.integer(Second),
    Film = factor(
      Video, 
      levels = seq_along(carma_files), 
      labels = str_extract(
        string = basename(carma_files), 
        pattern = "[^_]+"
      )
    ),
    .before = 1
  )
# Reshape and sort data
carma_tidy <- 
  carma_raw |> 
  pivot_longer(
    cols = starts_with("MW"), 
    names_to = "Rater", 
    values_to = "Rating",
    names_transform = parse_number
  ) |> 
  mutate(Rater = as.integer(Rater)) |> 
  arrange(Video, Rater, Second) |> 
  select(Film, Video, Rater, Second, Rating)
# Write data as a compressed RDS file
write_rds(carma_tidy, file = "./data/ratings_tidy.rds", compress = "gz")

Prepare holistic ratings

redcap_raw <- 
  read_xlsx("./data/redcap_cleaned.xlsx") |>
  separate(
    participant_id, 
    into = c("rater", "session"), 
    sep = "[_\\s]", 
    extra = "merge"
  ) |> 
  select(-starts_with("clip")) |> 
  pivot_longer(
    -c(rater, session), 
    names_to = "var", 
    values_to = "val"
  ) |> 
  separate(
    var, 
    into = c("clip", "scale"), 
    sep = "_", 
    extra = "merge"
  ) |> 
  filter(
    scale %in% c(
      "inspired", "alert", "excited", "enthusiastic", "determined", 
      "afraid", "upset", "nervous", "scared", "distressed"
    )
  ) |> 
  group_by(rater, clip, scale) |> 
  summarize(
    val = do.call("coalesce", as.list(val)),
    .groups = "drop"
  ) |> 
  mutate(
    scale = str_to_title(scale),
    clip = factor(clip, levels = c("aatb", "bs", "cmiyc", "dos", "f", "fg", "gm", "gwh", "ks", "lb", "lb2", "lms", "m", "mi", "ms", "ncom", "pf", "poh", "pt", "sn", "sotl", "z"),
                  labels = c("Akeelah", "BealeStreet", "CatchMe", "DaysSummer", "Fences", "ForrestGump", "GreenMile", "GoodWill", "KingsSpeech", "LadyBird", "LegallyBlonde", "LittleMiss", "Moonlight", "Miracle", "MarriageStory", "NoCountry", "PulpFiction", "Happyness", "ParentTrap", "SocialNetwork", "Silence", "Zodiac")),
  ) |> 
  print()
